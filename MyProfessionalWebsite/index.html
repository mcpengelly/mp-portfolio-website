                                    JScrollPane oScroller = new JScrollPane(incoming);
                                    oScroller.setVerticalScrollBarPolicy(ScrollPaneConstants.VERTICAL_SCROLLBAR_AS_NEEDED);
                                    oScroller.setHorizontalScrollBarPolicy(ScrollPaneConstants.HORIZONTAL_SCROLLBAR_NEVER);
                                    mainPanel.add(oScroller);

                                    //controls
                                    JButton sendButton = new JButton("Send");
                                    SendButtonListener sendBtnListener = new SendButtonListener();
                                    sendButton.addActionListener(sendBtnListener);
                                    mainPanel.add(sendButton);

                                    setupNetworking();

                                    //create a thread for listening to incoming data
                                    Thread readerThread = new Thread(new IncomingReader());
                                    readerThread.start();

                                    //frame setters
                                    frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
                                    frame.setVisible(true);
                                    frame.setSize(800, 600);



                                }

                        class ClientHandler implements Runnable {

                            Socket socket;

                            public ClientHandler(Socket clientSocket) {
                                try {
                                    socket = clientSocket;
                                    //chain that shit
                                    InputStreamReader isr = new InputStreamReader(socket.getInputStream());
                                    reader = new BufferedReader(isr);
                                } catch (Exception ex) {
                                    ex.printStackTrace();
                                }
                            }@Override
                            public void run() {
                                String message;
                                try {
                                    while ((message = reader.readLine()) != null) {
                                        System.out.println(message);
                                        tellEveryone(message);
                                    }
                                } catch (Exception ex) {
                                    ex.printStackTrace();
                                }
                            }
                            public void tellEveryone(String message) {
                                Iterator it = clientOutputStreams.iterator();
                                while (it.hasNext()) {
                                    try {
                                        PrintWriter writer = (PrintWriter) it.next();

                                        writer.flush();

                                        e.printStackTrace();
                                    }
                                }
